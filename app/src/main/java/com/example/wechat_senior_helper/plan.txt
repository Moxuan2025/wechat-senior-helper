按这两个点改。

## 1）发送语音：直接复用测试按钮那条稳定路径

不要再让正式语音流程自己做一套复杂监测。
把正式入口改成和测试按钮一致：**只负责“进入聊天后直接长按录音”**。

### 改动点

把 `WeChatVoiceFlow.sendVoiceMessage()` 收窄成稳定执行版：

```kotlin
class WeChatVoiceFlow(
    private val input: InputController
) {
    suspend fun sendVoiceMessage(durationMs: Long = 2000L): Boolean {
        // 这里不再做复杂监测，直接复用测试按钮的稳定执行方式
        input.longPressByRatio(0.5f, 0.99f, durationMs)
        return true
    }
}
```

### 语音模式判断只保留在“必要时”

如果你还想保留一个轻量检查，只做一次即可，不要切来切去：

```kotlin
suspend fun ensureVoiceModeIfNeeded(): Boolean {
    return isVoiceModeActive() || switchToVoiceMode()
}
```

然后正式流程里：

```kotlin
if (!voiceFlow.ensureVoiceModeIfNeeded()) return false
delay(200)
voiceFlow.sendVoiceMessage()
```

### 关键原则

你已经验证过 `requestSendVoice(2000L)` 这条链是稳定的，就让正式流程走同一条逻辑，不再加额外状态推断。

## 2）打电话：明确区分“语音电话 / 视频电话 / 只说打电话”

### 规则

* 说了“语音电话” → 直接 `AUDIO`
* 说了“视频电话” → 直接 `VIDEO`
* 只说“打电话” → **先确认**，不要直接执行

## 4）本地动作识别：把“打电话”拆成可确认和已明确两类

```kotlin
private fun parseActionOnly(text: String): ActionType {
    return when {
        text.contains("视频电话") || text.contains("视频通话") -> ActionType.VIDEO_CALL
        text.contains("语音电话") || text.contains("语音通话") -> ActionType.AUDIO_CALL
        text.contains("打电话") -> ActionType.AUDIO_CALL
        text.contains("发语音") -> ActionType.VOICE_MESSAGE
        text.contains("确认") -> ActionType.CONFIRM
        text.contains("取消") -> ActionType.CANCEL
        else -> ActionType.UNKNOWN
    }
}
```

这里先不改变你现有大框架，核心是：
**“打电话”本地只能识别为 `AUDIO_CALL`，但后续是否直接执行，要看文本有没有明确语音/视频。**

## 6）`handleByLLM()` 改成：打电话默认先确认

### 修改核心逻辑

```kotlin
suspend fun handleByLLM(userText: String, hintedAction: ActionType): RouteResult {
    val rawResult = llmParseIntent(userText, hintedAction)
    val result = repairLlmResult(rawResult, hintedAction)

    val target = normalizeTargetFromLLM(result.target ?: "")
    if (target.isBlank()) {
        return RouteResult.Reply("请告诉我要操作谁")
    }

    return when (result.actionType) {
        ActionType.VOICE_MESSAGE -> {
            if (result.needConfirm) {
                RouteResult.NeedConfirm(
                    PendingAction.Voice(target),
                    "要给$target发语音吗？"
                )
            } else {
                RouteResult.DirectExecute(
                    action = {
                        if (!chatNavGuard.ensureChatWithTarget(target)) return@DirectExecute false
                        delay(200)
                        voiceFlow.sendVoiceMessage()
                    },
                    message = "好的，正在给$target发语音..."
                )
            }
        }

        ActionType.AUDIO_CALL -> {
            // 只要不是“语音电话”这种明确表达，就先确认
            val explicitAudio = userText.contains("语音电话") || userText.contains("语音通话")
            if (!explicitAudio) {
                return RouteResult.NeedConfirm(
                    PendingAction.Call(target, CallMode.UNKNOWN),
                    "要给$target打语音电话还是视频电话？"
                )
            }

            if (result.needConfirm) {
                RouteResult.NeedConfirm(
                    PendingAction.Call(target, CallMode.AUDIO),
                    "要给$target打语音电话吗？"
                )
            } else {
                RouteResult.DirectExecute(
                    action = {
                        if (!chatNavGuard.ensureChatWithTarget(target)) return@DirectExecute false
                        delay(800)
                        callFlow.makeCall(CallMode.AUDIO)
                    },
                    message = "好的，正在给$target打语音电话..."
                )
            }
        }

        ActionType.VIDEO_CALL -> {
            if (result.needConfirm) {
                RouteResult.NeedConfirm(
                    PendingAction.Call(target, CallMode.VIDEO),
                    "要给$target打视频电话吗？"
                )
            } else {
                RouteResult.DirectExecute(
                    action = {
                        if (!chatNavGuard.ensureChatWithTarget(target)) return@DirectExecute false
                        delay(800)
                        callFlow.makeCall(CallMode.VIDEO)
                    },
                    message = "好的，正在给$target打视频电话..."
                )
            }
        }

        else -> RouteResult.Reply("我没理解清楚，请再说一遍")
    }
}
```

## 7）确认按钮支持“电话类型二选一”

你现在的 `PendingAction.Call(target, CallMode.UNKNOWN)` 要能进入一个二次确认状态。

```kotlin
btnConfirm.setOnClickListener {
    val action = pendingAction ?: return@setOnClickListener

    when (action) {
        is PendingAction.Call -> {
            if (action.mode == CallMode.UNKNOWN) {
                showMessage("请确认：语音电话还是视频电话？")
                showCallModeChooser(action.target)
                return@setOnClickListener

        else -> {}

    hideFloatingBall()

        val ok = service.confirmAction(action)

        showFloatingBall()
    }
}
```

## 8）`confirmAction()` 里补上 UNKNOWN 分支

```kotlin
suspend fun confirmAction(action: PendingAction): Boolean {
    return when (action) {
        is PendingAction.Voice -> {
            if (!chatNavGuard.ensureChatWithTarget(action.target)) return false
            delay(200)
            voiceFlow.sendVoiceMessage()
        }

        is PendingAction.Call -> {
            val mode = action.mode
            if (mode == CallMode.UNKNOWN) return false
            if (!chatNavGuard.ensureChatWithTarget(action.target)) return false
            delay(800)
            callFlow.makeCall(mode)
        }
    }
}
```
